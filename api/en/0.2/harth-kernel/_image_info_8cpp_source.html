123  bool ImageInfo::Load()
124  {
125  m_macho = new MachO(m_header);
126  if (!m_macho->IsValid())
127  {
128  StringPrintf(m_error, "Image \"%s\" has invalid MachO header", m_name);
129  return false;
130  }
131 
132  MemRef cmdIter;
133  while (MemRef cmd = m_macho->NextLoadCommand(cmdIter))
134  {
135  const load_command &lc = cmd.As<const load_command>();
136  switch (lc.cmd)
137  {
138  case LC_SEGMENT_64:
139  if (!LoadSegment(cmd))
140  return false;
141  break;
142 
143  case LC_SYMTAB:
144  if (!LoadSymTab(cmd))
145  return false;
146  if (!LoadObjects())
147  return false;
148  break;
149  }
150  }
151  return true;
152  }

202  bool ImageInfo::LoadObjects()
203  {
204  Object *obj = nullptr;
205 
206  for (size_t i = 0; i < m_symtab->GetCount(); i++)
207  {
208  const nlist_64 &nl = m_symtab->GetEntry(i);
209  switch (nl.n_type)
210  {
211  case N_OSO:
212  m_objects.emplace_back();
213  obj = &(m_objects.back());
214  obj->path = m_symtab->GetString(nl);
215  break;
216 
217  case N_FUN:
218  if (obj == nullptr)
219  {
220  StringPrintf(m_error, "Symbol table has N_FUN before N_OSO in \"%s\"",
221  m_name);
222  return false;
223  }
224  if (nl.n_sect)
225  {
226  Function fn;
227  fn.symbol = m_symtab->GetString(nl);
228  fn.start = nl.n_value;
229  fn.end = nl.n_value;
230  obj->functions.push_back(fn);
231  break;
232  }
233 
234  if (obj->functions.size() == 0)
235  {
236  StringPrintf(m_error, "Symbol table has N_FUN end before begin in \"%s\"",
237  m_name);
238  return false;
239  }
240  Function &fn = obj->functions.back();
241  fn.end += nl.n_value;
242  break;
243  }
244  }
245 
246  return true;

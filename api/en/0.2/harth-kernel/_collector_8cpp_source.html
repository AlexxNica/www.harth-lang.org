146  void Collector::MarkAddTaskToGreySet(const OS::Task &task)
147  {
148  // Mark stack
149  // TODO: Is this portable enough?
150  const OS::Stack &stack = task.GetStack();
151  auto stackMem = stack.GetUsed().AsArray<const byte_t>();
152  const byte_t *stackBegin = stackMem.GetPointerBegin();
153  const byte_t *stackEnd = stackMem.GetPointerEnd();
154 
155  HK_ASSERT(Util::IsPointerAligned(stackBegin, OS::Stack::GetStackAlignment()));
156  HK_ASSERT(Util::IsPointerAligned(stackEnd, OS::Stack::GetStackAlignment()));
157  HK_ASSERT(Util::IsPointerAligned(stackBegin, OS::Stack::GetPointerOnStackAlignment()));
158  HK_ASSERT(Util::IsPointerAligned(stackEnd, OS::Stack::GetPointerOnStackAlignment()));
159 
160  for (const byte_t *stackPtr = stackBegin;
161  stackPtr < stackEnd;
162  stackPtr += OS::Stack::GetPointerOnStackAlignment())
163  {
164  const Object *objectPtr = reinterpret_cast<const Object *>(stackPtr);
165 
166  if (ObjectPool::IsMarkableConservative(*objectPtr))
167  MarkPointer(objectPtr);
168  }
169 
170  // Mark registers
171  // TODO: Is this portable enough?
172  OS::Task::Register registers[OS::Task::GetRegisterCount()];
173  task.GetRegisters(registers, sizeof(registers));
174 
175  for (size_t i = 0; i < OS::Task::GetRegisterCount(); i++)
176  {
177  const Object *objectPtr = reinterpret_cast<const Object *>(registers[i]);
178 
179  if (ObjectPool::IsMarkableConservative(*objectPtr))
180  MarkPointer(objectPtr);
181  }
182  }

184  void Collector::MarkGreySet()
185  {
186  while (m_greySet.IsNotEmpty())
187  {
188  const Object *objPtr = m_greySet.RemovePointer();
189  // TODO: Ensure obj is valid (null is valid but is not marked)
190  // TODO: Should we check for nullptr in Collector::MarkPointer()?
191  // TODO: Others must be valid pointer; perhaps only validation, and
192  // stack sweep should filter.
193  if (objPtr == nullptr)
194  continue;
195 
196  const Object &obj = *objPtr;
197  ObjectBlock &objectBlock = ObjectPool::GetObjectBlockFromAnyDataPointer(&obj);
198  if (objectBlock.IsMarked(obj))
199  continue;
200 
201  objectBlock.AddMark(obj);
202  obj.GCMarkPointers(*this);
203  }
204  }

232  void Collector::SweepClass(Class &klass)
233  {
234  const ObjectPool &objectPool = klass.GetObjectPool();
235  for (ObjectPoolIterator it(objectPool); !it.IsEnd(); it.Next())
236  {
237  if (!it.IsFree())
238  {
239  if (!it.IsMarked())
240  DestructAndFreeObject(klass, it);
241  }
242 
243  if (it.IsLastObjectInBlock())
244  {
245  ObjectBlock *objectBlock = it.GetMutableObjectBlockPointer();
246  objectBlock->RemoveAllMarks();
247  }
248  }
